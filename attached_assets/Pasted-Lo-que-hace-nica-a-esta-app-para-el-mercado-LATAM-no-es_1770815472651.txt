3. JavaScript (El Cerebro Comercial)
Este c√≥digo gestiona:

Persistencia: Guarda todo en la memoria del navegador (localStorage), as√≠ que si cierras y abres, los datos siguen ah√≠ (como una app real).

L√≥gica Financiera: Calcula el presupuesto diario bas√°ndose en los d√≠as que faltan.

JavaScript
// === BASE DE DATOS LOCAL ===
const db = {
    config: JSON.parse(localStorage.getItem('flux_config')) || null,
    transactions: JSON.parse(localStorage.getItem('flux_transactions')) || []
};

// Categor√≠as "Latam Standard"
const categories = [
    { id: 'super', name: 'Super', icon: 'üõí' },
    { id: 'delivery', name: 'Delivery', icon: 'üçî' },
    { id: 'transporte', name: 'Transp.', icon: 'üöå' },
    { id: 'ocio', name: 'Ocio', icon: 'üçª' },
    { id: 'servicios', name: 'Servicios', icon: 'üí°' },
    { id: 'varios', name: 'Varios', icon: 'üì¶' }
];

let selectedCat = '';

// === INICIALIZACI√ìN ===
function init() {
    if (!db.config) {
        document.getElementById('screen-setup').style.display = 'flex';
    } else {
        updateDashboard();
        renderHistory();
    }
    renderCategories();
}

// === L√ìGICA DE CONFIGURACI√ìN ===
function saveSetup() {
    const income = parseFloat(document.getElementById('setup-income').value) || 0;
    const debt = parseFloat(document.getElementById('setup-debt').value) || 0;
    const fixed = parseFloat(document.getElementById('setup-fixed').value) || 0;

    // Guardamos la configuraci√≥n
    db.config = {
        baseIncome: income,
        preDeducted: debt, // Adelantos
        fixedCosts: fixed, // Alquiler + Cuotas
        startDate: new Date().toISOString()
    };

    localStorage.setItem('flux_config', JSON.stringify(db.config));
    document.getElementById('screen-setup').style.display = 'none';
    updateDashboard();
}

// === L√ìGICA DEL DASHBOARD (VEREDICTO) ===
function updateDashboard() {
    // 1. Calcular Saldo Inicial Real
    // F√≥rmula: (Ingreso - Adelantos - Fijos)
    const initialAvailable = db.config.baseIncome - db.config.preDeducted - db.config.fixedCosts;

    // 2. Sumar Gastos del Mes
    const totalSpent = db.transactions.reduce((acc, curr) => acc + curr.amount, 0);

    // 3. Dinero Actual en Mano
    const currentBalance = initialAvailable - totalSpent;

    // 4. D√≠as Restantes (Asumimos fin de mes para el ejemplo, pero puede ser configurable)
    const today = new Date();
    const lastDay = new Date(today.getFullYear(), today.getMonth() + 1, 0); // √öltimo d√≠a del mes
    const daysLeft = Math.max(1, lastDay.getDate() - today.getDate());

    // 5. Presupuesto Diario (Veredicto)
    const dailyBudget = Math.floor(currentBalance / daysLeft);

    // --- ACTUALIZAR UI ---
    document.getElementById('total-balance').innerText = `$${currentBalance.toLocaleString()}`;
    document.getElementById('total-spent').innerText = `$${totalSpent.toLocaleString()}`;
    document.getElementById('days-left').innerText = `${daysLeft} D√çAS RESTANTES`;
    document.getElementById('daily-budget').innerText = `$${dailyBudget}`;

    // Color del Gauge seg√∫n salud financiera
    const gaugeFill = document.getElementById('gauge-fill');
    let color = '#10b981'; // Verde
    let percent = 100;

    // Si gastaste m√°s del 80% de tu disponible real
    const spentPercentage = (totalSpent / initialAvailable) * 100;
    
    if (currentBalance < 0) {
        color = '#ef4444'; // Rojo (Deuda)
        percent = 100;
        document.getElementById('daily-budget').innerText = '$0'; // No puedes gastar
    } else if (spentPercentage > 75) {
        color = '#f59e0b'; // Amarillo (Cuidado)
        percent = 100 - spentPercentage;
    } else {
        percent = 100 - spentPercentage;
    }

    gaugeFill.style.background = `conic-gradient(${color} ${percent}%, transparent ${percent}%)`;
}

// === AGREGAR GASTOS ===
function renderCategories() {
    const grid = document.getElementById('cat-grid');
    grid.innerHTML = '';
    categories.forEach(cat => {
        grid.innerHTML += `
            <div class="cat-btn" onclick="selectCat(this, '${cat.id}')">
                <span style="font-size:24px">${cat.icon}</span>
                <span style="font-size:12px">${cat.name}</span>
            </div>
        `;
    });
}

function selectCat(el, id) {
    document.querySelectorAll('.cat-btn').forEach(btn => btn.classList.remove('selected'));
    el.classList.add('selected');
    selectedCat = id;
}

function addTransaction() {
    const amount = parseFloat(document.getElementById('new-amount').value);
    if (!amount || !selectedCat) {
        alert("Ingresa monto y categor√≠a");
        return;
    }

    const newTx = {
        id: Date.now(),
        amount: amount,
        category: selectedCat,
        date: new Date().toISOString(),
        note: ''
    };

    db.transactions.unshift(newTx); // Agregar al inicio
    localStorage.setItem('flux_transactions', JSON.stringify(db.transactions));
    
    closeAdd();
    document.getElementById('new-amount').value = '';
    updateDashboard();
    renderHistory();
}

// === HISTORIAL ===
function renderHistory() {
    const list = document.getElementById('recent-tx-list');
    const fullList = document.getElementById('full-history-list');
    
    // Generar HTML
    let html = '';
    if (db.transactions.length === 0) {
        html = '<div style="text-align:center; padding:20px; color:#64748b">Sin movimientos a√∫n</div>';
    } else {
        db.transactions.forEach(tx => {
            const catInfo = categories.find(c => c.id === tx.category) || { name: 'Gasto', icon: 'üí∏' };
            const date = new Date(tx.date).toLocaleDateString('es-UY', { day: 'numeric', month: 'short' });
            
            html += `
                <div class="tx-item">
                    <div class="tx-left">
                        <div class="tx-icon">${catInfo.icon}</div>
                        <div class="tx-info">
                            <h4>${catInfo.name}</h4>
                            <span>${date}</span>
                        </div>
                    </div>
                    <div class="tx-amount neg">-$${tx.amount}</div>
                </div>
            `;
        });
    }

    if(list) list.innerHTML = html; // En dashboard (solo mostrar√© los primeros 3 con CSS si quisiera limitar)
    if(fullList) fullList.innerHTML = html; // En pantalla historial
}
